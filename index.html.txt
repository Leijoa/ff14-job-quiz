            { id: 'MIN', name: '採掘師', type: 'dol' },

            { id: 'FSH', name: '漁師', type: 'dol' }
        ];

        // 遊戲邏輯控制器
        const game = {
            round: 0,
            score: 0,
            totalRounds: 10,
            currentAnswer: null,
            isLocked: false, // 防止重複點擊

            // DOM 元素快取
            el: {
                start: document.getElementById('start-screen'),
                game: document.getElementById('game-area'),
                end: document.getElementById('end-screen'),
                icon: document.getElementById('question-icon'),
                options: document.getElementById('options-container'),
                progress: document.getElementById('progress'),
                score: document.getElementById('score'),
                finalScore: document.getElementById('final-score'),
                comment: document.getElementById('comment')
            },

            start: function() {
                this.round = 0;
                this.score = 0;
                this.el.start.style.display = 'none';
                this.el.end.style.display = 'none';
                this.el.game.style.display = 'block';
                this.nextRound();
            },

            nextRound: function() {
                if (this.round >= this.totalRounds) {
                    this.finish();
                    return;
                }

                this.round++;
                this.isLocked = false;
                this.updateStats();

                // 1. 隨機抽選一題
                this.currentAnswer = fullJobsData[Math.floor(Math.random() * fullJobsData.length)];
                
                // 設定圖片路徑 (假設放在 icons 資料夾，且檔名為 ID.png)
                this.el.icon.src = `icons/${this.currentAnswer.id}.png`;

                // 2. 產生混淆選項
                let options = [this.currentAnswer];
                
                // 嘗試加入同類型的混淆項 (讓難度變高)
                // 例如：如果是採集，儘量找其他採集來混淆
                const sameTypeJobs = fullJobsData.filter(j => j.type === this.currentAnswer.type && j.id !== this.currentAnswer.id);
                const otherJobs = fullJobsData.filter(j => j.type !== this.currentAnswer.type);
                
                // 邏輯：優先從同類型抓 1-2 個，剩下隨機補
                while (options.length < 4) {
                    let pool = (options.length < 3 && sameTypeJobs.length > 0) ? sameTypeJobs : fullJobsData;
                    let randomJob = pool[Math.floor(Math.random() * pool.length)];
                    
                    if (!options.find(o => o.id === randomJob.id)) {
                        options.push(randomJob);
                    }
                }

                // 3. 洗牌
                options.sort(() => Math.random() - 0.5);

                // 4. 渲染按鈕
                this.el.options.innerHTML = '';
                options.forEach(job => {
                    const btn = document.createElement('button');
                    btn.textContent = job.name;
                    // 使用 closure 傳遞參數
                    btn.onclick = (e) => this.check(job, e.target);
                    this.el.options.appendChild(btn);
                });
            },

            check: function(selectedJob, btnElement) {
                if (this.isLocked) return;
                this.isLocked = true; // 鎖定防止連點

                const buttons = this.el.options.querySelectorAll('button');
                const isCorrect = selectedJob.id === this.currentAnswer.id;

                if (isCorrect) {
                    btnElement.classList.add('correct');
                    this.score++;
                } else {
                    btnElement.classList.add('wrong');
                    // 標示正確答案給玩家看
                    buttons.forEach(b => {
                        if (b.textContent === this.currentAnswer.name) {
                            b.classList.add('correct');
                        }
                    });
                }

                // 延遲進入下一題
                setTimeout(() => {
                    this.nextRound();
                }, 1200); // 1.2秒讓玩家看清楚答案
            },

            updateStats: function() {
                this.el.progress.textContent = `題數: ${this.round}/${this.totalRounds}`;
                this.el.score.textContent = `得分: ${this.score}`;
            },

            finish: function() {
                this.el.game.style.display = 'none';
                this.el.end.style.display = 'block';
                this.el.finalScore.textContent = `${this.score} / ${this.totalRounds}`;
                
                // 根據 ENTP 與玩家屬性客製化評語
                let msg = "";
                if (this.score === 10) {
                    msg = "太神啦！你的靈魂水晶是不是全套齊全？";
                } else if (this.score >= 8) {
                    msg = "非常優秀！艾歐澤亞的百科全書就是你。";
                } else if (this.score >= 5) {
                    msg = "還不錯，但小心別把刻木匠認成木工喔！";
                } else {
                    msg = "看來你需要去死者宮殿特訓一下基礎職業了...";
                }
                this.el.comment.textContent = msg;
            }
        };
